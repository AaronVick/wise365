// Step 2: Define helper functions
async function getAgentIdMap() {
  const agents = await prisma.agent.findMany();
  const agentIdMap = {};
  agents.forEach((agent) => {
    agentIdMap[agent.name] = agent.id;
  });
  return agentIdMap;
}

// Step 3: Seed the AgentData table
async function seedAgentData() {
  const agentIdMap = await getAgentIdMap();

  for (const data of rawAgentData) {
    const agentId = agentIdMap[data.agentId];
    if (!agentId) {
      console.error(`Agent ID not found for agent name: ${data.agentId}`);
      continue;
    }

    await prisma.agentData.create({
      data: {
        agentId,
        dataType: data.dataType,
        description: data.description,
        content: data.content,
      },
    });
  }

  console.log('Agent data seeding complete!');
}

// Execute the seeding logic
seedAgentData()
  .catch((e) => {
    console.error(e);
    process.exit(1);
  })
  .finally(async () => {
    await prisma.$disconnect();
  });
